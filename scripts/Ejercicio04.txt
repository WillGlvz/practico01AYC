    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
        int ultima = jTextField1.getText().length() -1;
        int penultima = jTextField1.getText().length() -2;
        if(jTextField1.getText().charAt(0) == 'a' && jTextField1.getText().charAt(1) == 'c'){
            JOptionPane.showMessageDialog(rootPane, "La cadena ingresada es correcta", "Atención", JOptionPane.INFORMATION_MESSAGE);
        }else if (jTextField1.getText().charAt(penultima) == 'a' && jTextField1.getText().charAt(ultima) == 'b'){
            JOptionPane.showMessageDialog(rootPane, "La cadena ingresada es correcta", "Atención", JOptionPane.INFORMATION_MESSAGE);
        }else{
            JOptionPane.showMessageDialog(rootPane, "La cadena ingresada es inválida", "Atención", JOptionPane.CANCEL_OPTION);
        }
    }//GEN-LAST:event_jButton1ActionPerformed
